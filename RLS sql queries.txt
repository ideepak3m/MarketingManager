RLS updates for Campaigns table
CREATE POLICY campaigns_update_by_id ON public.campaigns
  FOR UPDATE
  TO authenticated
  USING (
    user_id IN (
      SELECT nus.session_id
      FROM public.nova_user_sessions nus
      WHERE nus.user_id = auth.uid()
    )
  )
  WITH CHECK (
    user_id IN (
      SELECT nus.session_id
      FROM public.nova_user_sessions nus
      WHERE nus.user_id = auth.uid()
    )
  );
